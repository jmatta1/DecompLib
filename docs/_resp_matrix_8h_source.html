154 template<typename ParamType>
155 void RespMatrix<ParamType>::calculateSummedRows()
156 {
157  for(int i=0; i<numFunc; ++i)
158  {
159  int offset = i*respLen;
160  int edge = (offset+respLen);
161  summedRows[i] = 0.0;
162  ParamType comp = 0.0;
163  for(int j=offset; j<edge; ++j)
164  {
165  ParamType compInput = (matrix[j] - comp);
166  ParamType tempSum = (summedRows[i] + compInput);
167  comp = ((tempSum - summedRows[i]) - compInput);
168  summedRows[i] += matrix[j];
169  }
170  }
171  rowSumPerformed = true;
172 }
