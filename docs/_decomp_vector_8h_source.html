145 void DecompVector<ParamType>::initWithDataVector(const DataVector<ParamType>& val, const ParamType& factor, const ParamType& offset)
146 {
147  int valSize = val.getLength();
148  assert(valSize >= size);
149  ParamType stepSize = static_cast<ParamType>(valSize) / static_cast<ParamType>(size);
150  for(int i=0; i<size; ++i)
151  {
152  int index = static_cast<int>(i*stepSize);
153  if(index < valSize)
154  {
155  vec[i] = ((factor*val.getElement(index))+offset);
156  //safety net to handle cases where the input spectrum is zero
157  if(vec[i] < TinyValue)
158  {
159  vec[i] = TinyValue;
160  }
161  }
162  else
163  {
164  vec[i] = ((factor*val.getElement(valSize-1))+offset);
165  //safety net to handle cases where the input spectrum is zero
166  if(vec[i] < TinyValue)
167  {
168  vec[i] = TinyValue;
169  }
170  }
171 
172  }
173 }
