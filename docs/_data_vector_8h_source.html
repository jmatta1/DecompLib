91 bool DataVector<ParamType>::isSafe()
92 {
93  //initially assume the input spectrum is safe
94  bool output = true;
95  //since the input spectrum is allowed to have some zeros, check to make sure that
96  //the *entire* vector is not zero using kahan summation (while also checking
97  //to be certain that there are no negative values)
98  ParamType sum = 0.0;
99  ParamType comp = 0.0;
100  for(int i=0; i<size; ++i)
101  {
102  if(vec[i] < static_cast<ParamType>(0.0))
103  {
104  if(printErrors) std::cout<<"Bin number "<<i<<" in input spectrum is negative"<<std::endl;
105  output = false;
106 
107  }
108  ParamType compInput = (vec[i] - comp);
109  ParamType tempSum = (sum + compInput);
110  comp = ((tempSum - sum) - compInput);
111  sum = tempSum;
112  }
113  if(sum <= static_cast<ParamType>(0.0))//we have a zero or negative vector
114  {
115  if(printErrors) std::cout<<"Sum of all bins in input spectrum was zero or negative"<<std::endl;
116  output = false;
117  }
118 
119  return output;
120 }
